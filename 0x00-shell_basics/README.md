#pwd
Print current working directory

#ls
list the file contents of current directory

#cd .. 
Change the current directory to the home directory

#ls -l
List file contents of current directory in a long format

#ls -la
List file contents incuding hidden files of current directory in a long format

#ls -na
List file contents including hidden files of current directory showing their groupId and UserId numerically
